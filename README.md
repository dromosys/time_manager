# setup
 1. git clone https://github.com/dromosys/time_manager
 1. cd time_manager
 1. python3 manage.py makemigrations
 1. python3 manage.py migrate
 1. python3 manage.py createsuperuser
 1. python3 manage.py runserver
 1. sqlite3 db/db.sqlite3
